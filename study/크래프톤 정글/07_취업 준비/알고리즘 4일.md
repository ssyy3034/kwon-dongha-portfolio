```python
# BFS 템플릿
from collections import deque
def bfs(start):
    visited = set([start])
    q = deque([start])
    while q:
        cur = q.popleft()
        for nxt in graph[cur]:
            if nxt not in visited:
                visited.add(nxt)
                q.append(nxt)

# DFS 템플릿 (스택)
def dfs(start):
    visited = set()
    stack = [start]
    while stack:
        cur = stack.pop()
        if cur in visited:
            continue
        visited.add(cur)
        for nxt in graph[cur]:
            stack.append(nxt)

# 2차원 격자 BFS
```

```python
# 기본 이분탐색
def binary_search(arr, target):
    lo, hi = 0, len(arr) - 1
    while lo <= hi:
        mid = (lo + hi) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            lo = mid + 1
        else:
            hi = mid - 1
    return -1

# Parametric Search (조건 만족하는 최솟값/최댓값)
lo, hi = min_val, max_val
while lo <= hi:
    mid = (lo + hi) // 2
    if check(mid):
        answer = mid
        hi = mid - 1  # 최솟값 찾기 (최댓값이면 lo = mid + 1)
    else:
        lo = mid + 1
```
